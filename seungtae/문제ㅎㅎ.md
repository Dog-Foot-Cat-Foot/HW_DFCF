### 다음 중 오류가 나는 코드를 모두 고르시오
1. 
<script>
  function num ()
  {
    var sum = "Hello World!";
    return sum;
  }
  num();
</script>



2. 
<script>
  function num ()
  {
    var sum = "Hello World!";
  }
  num();
</script>



3. 
<script>
  function plus ( A, B )
  {
    var sum = A + B;
    document.write(sum);
  }
  plus ("Hello", "World!");
</script>



4. 
<script>
  function num ()
  {
    document.write("Hello World!");
  }
  num();
</script>
