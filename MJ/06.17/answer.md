select
    cus_no        "고객번호"
    ,cus_name     "고객명"
    ,to_char(to_date(case
                    when substr(cus_jumin,7,1)='1' then '19'
                    when substr(cus_jumin,7,1)='2' then '19'
                    else '20'
            end||substr(cus_jumin,1,6),'yyyymmdd'),'yyyy.mm.dd') "생년월일"
    ,decode(substr(cus_jumin,7,1),'1','남','3','남','여')   "성별"
from
    customer;
